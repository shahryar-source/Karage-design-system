Toast Notifications

Floating toasts — positioned top-right. Click the buttons to trigger them.

JS 
function showToast(type, icon, message) {
  const toast = document.createElement('div');
  toast.className = `k-toast k-toast-${type}`;
  toast.innerHTML = `<span>${icon}</span><span>${message}</span>`;
  document.getElementById('toast-container').appendChild(toast);
  setTimeout(() => toast.remove(), 3000);
}

/* CSS */
#toast-container {
  position: fixed; top: 20px; right: 20px; z-index: 9999;
  display: flex; flex-direction: column; gap: 10px;
}
.k-toast {
  display: flex; align-items: center; gap: 10px;
  padding: 12px 18px; border-radius: var(--radius-md-10);
  box-shadow: 0 8px 24px rgba(0,0,0,0.12);
  font-size: 14px; font-weight: 500;
  animation: slideIn 0.25s ease;
}
.k-toast-success { background: #134F1A; color: #B6F174; }
.k-toast-warning { background: #8E6A0C; color: #FDE2B0; }
.k-toast-error   { background: #D92D20; color: #FEE4E2; }
.k-toast-info    { background: #1A3FBF; color: #DBE5FF; }
@keyframes slideIn {
  from { transform: translateX(40px); opacity: 0; }
  to   { transform: translateX(0); opacity: 1; }
}
